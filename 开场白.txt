function splitSpirits(text) {
    var value = (text || '').trim();
    var list = value ? value.split(/[、,，;；\s]+/) : [];
    list = list.map(function(item) { return item.trim(); }).filter(function(item) { return item !== ''; });
    return list.length ? list.slice(0, 2) : ['未觉醒'];
}

function readStorageJSON(key) {
    try {
        var raw = localStorage.getItem(key);
        if (!raw) return {};
        var parsed = JSON.parse(raw);
        return (parsed && typeof parsed === 'object') ? parsed : {};
    } catch (err) {
        return {};
    }
}

function writeStorageJSON(key, value) {
    try {
        localStorage.setItem(key, JSON.stringify(value || {}));
        return true;
    } catch (err) {
        return false;
    }
}

function loadTemplateStore() {
    return readStorageJSON(FORM_TEMPLATE_STORAGE_KEY);
}

function saveTemplateStore(store) {
    return writeStorageJSON(FORM_TEMPLATE_STORAGE_KEY, store || {});
}

function collectFormSnapshot() {
    var destiny = 'A';
    document.querySelectorAll('input[name="destiny"]').forEach(function(radio) {
        if (radio.checked) destiny = radio.value;
    });

    var spiritTypes = [];
    var spiritDescs = [];
    document.querySelectorAll('.spirit-type-select').forEach(function(select) {
        spiritTypes.push(select.value || '');
    });
    document.querySelectorAll('.spirit-desc-input').forEach(function(input) {
        spiritDescs.push(input.value || '');
    });

    return {
        timeline: (document.getElementById('timeline-select') || {}).value || '13岁',
        name: (document.getElementById('char-name') || {}).value || '',
        gender: (document.getElementById('char-gender') || {}).value || '男',
        age: (document.getElementById('char-age') || {}).value || '13',
        loc: (document.getElementById('char-loc') || {}).value || '',
        spirit: (document.getElementById('char-spirit') || {}).value || '',
        personality: (document.getElementById('char-personality') || {}).value || '',
        appearance: (document.getElementById('char-appearance') || {}).value || '',
        talent: (document.getElementById('char-talent') || {}).value || 'EX',
        bg: (document.getElementById('char-bg') || {}).value || '',
        jobType: (document.getElementById('char-job-type') || {}).value || '无',
        jobLv: (document.getElementById('char-job-lv') || {}).value || '1',
        destiny: destiny,
        spiritTypes: spiritTypes,
        spiritDescs: spiritDescs
    };
}

function applyFormSnapshot(snapshot) {
    if (!snapshot || typeof snapshot !== 'object') return;

    function setValue(id, value) {
        var el = document.getElementById(id);
        if (!el || value === undefined || value === null) return;
        el.value = value;
    }

    setValue('timeline-select', snapshot.timeline);
    setValue('char-name', snapshot.name);
    setValue('char-gender', snapshot.gender);
    setValue('char-age', snapshot.age);
    setValue('char-loc', snapshot.loc);
    setValue('char-spirit', snapshot.spirit);
    setValue('char-personality', snapshot.personality);
    setValue('char-appearance', snapshot.appearance);
    setValue('char-talent', snapshot.talent);
    setValue('char-bg', snapshot.bg);
    setValue('char-job-type', snapshot.jobType);
    setValue('char-job-lv', snapshot.jobLv);

    if (typeof window.updateSpiritTypes === 'function') {
        window.updateSpiritTypes();
    }

    var spiritTypes = Array.isArray(snapshot.spiritTypes) ? snapshot.spiritTypes : [];
    var spiritDescs = Array.isArray(snapshot.spiritDescs) ? snapshot.spiritDescs : [];

    document.querySelectorAll('.spirit-type-select').forEach(function(select, index) {
        if (spiritTypes[index] !== undefined) {
            select.value = spiritTypes[index];
        }
    });
    document.querySelectorAll('.spirit-desc-input').forEach(function(input, index) {
        if (spiritDescs[index] !== undefined) {
            input.value = spiritDescs[index];
        }
    });

    var destiny = snapshot.destiny || 'A';
    var selected = document.querySelector('input[name="destiny"][value="' + destiny + '"]');
    if (selected) {
        var card = selected.closest ? selected.closest('.radio-card') : null;
        if (card && typeof window.selectRadio === 'function') {
            window.selectRadio(card);
        } else {
            selected.checked = true;
        }
    }
}

    var names = Object.keys(store).sort(function(a, b) {
        var at = (store[a] && store[a].updatedAt) || 0;
        var bt = (store[b] && store[b].updatedAt) || 0;
        if (bt !== at) return bt - at;
        return a > b ? 1 : (a < b ? -1 : 0);
    });

    select.innerHTML = '';
    var placeholder = document.createElement('option');
    placeholder.value = '';
    placeholder.textContent = '选择已保存模板';
    select.appendChild(placeholder);

    names.forEach(function(name) {
        var option = document.createElement('option');
        option.value = name;
        option.textContent = name;
        select.appendChild(option);
    });

    if (selectedName && store[selectedName]) {
        select.value = selectedName;
    }
}

function bindTemplateMemoryEvents() {
    var select = document.getElementById('template-select');
    if (select) {
        select.addEventListener('change', function() {

            if (nameInput && select.value) {
                nameInput.value = select.value;
            }
        });
    }

    var root = document.getElementById('character-creation-root');
    if (root) {
        var persistLast = function() {
            saveLastSnapshot(collectFormSnapshot());
        };
        root.addEventListener('input', persistLast);
        root.addEventListener('change', persistLast);
    }
}

window.saveCurrentTemplate = function() {
    var nameInput = document.getElementById('template-name');
    var name = (nameInput && nameInput.value ? nameInput.value : '').trim();
    if (!name) {
        showTemplateMessage('请先输入模板名，再点击保存。', true);
        return;
    }

    var store = loadTemplateStore();
    store[name] = {
        updatedAt: Date.now(),
        snapshot: collectFormSnapshot()
    };

    if (!saveTemplateStore(store)) {
        showTemplateMessage('保存失败：本地存储不可用。', true);
        return;
    }

    refreshTemplateSelect(name);
    showTemplateMessage('已保存模板：' + name);
};

window.loadSelectedTemplate = function() {
    var select = document.getElementById('template-select');
    var name = (select && select.value) ? select.value : '';
    if (!name) {
        showTemplateMessage('请先从列表选择模板。', true);
        return;
    }

    var store = loadTemplateStore();
    var entry = store[name];
    if (!entry || !entry.snapshot) {
        showTemplateMessage('模板不存在，列表已刷新。', true);
        refreshTemplateSelect();
        return;
    }

    applyFormSnapshot(entry.snapshot);
    saveLastSnapshot(entry.snapshot);

    var nameInput = document.getElementById('template-name');
    if (nameInput) nameInput.value = name;

    showTemplateMessage('已加载模板：' + name);
};

window.deleteSelectedTemplate = function() {

    var name = (select && select.value) ? select.value : '';
    if (!name) {
        showTemplateMessage('请先选择要删除的模板。', true);

};
